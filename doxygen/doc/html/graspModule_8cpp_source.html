58 /**********************************************************************/
59 Property GraspingModule::get_grasping_pose(const Property &estimated_superq, const string &hand_str)
60 {
61  //LockGuard lg(mutex);
62 
63  Bottle *dim=estimated_superq.find("dimensions").asList();
64 
65  if (!estimated_superq.find("dimensions").isNull())
66  {
67  object[0]=dim->get(0).asDouble(); object[1]=dim->get(1).asDouble(); object[2]=dim->get(2).asDouble();
68  }
69 
70  Bottle *shape=estimated_superq.find("exponents").asList();
71 
72  if (!estimated_superq.find("exponents").isNull())
73  {
74  object[3]=shape->get(0).asDouble(); object[4]=shape->get(1).asDouble();
75  }
76 
77  Bottle *exp=estimated_superq.find("exponents").asList();
78 
79  if (!estimated_superq.find("exponents").isNull())
80  {
81  object[3]=exp->get(0).asDouble(); object[4]=exp->get(1).asDouble();
82  }
83 
84  Bottle *center=estimated_superq.find("center").asList();
85 
86  if (!estimated_superq.find("center").isNull())
87  {
88  object[5]=center->get(0).asDouble(); object[6]=center->get(1).asDouble(); object[7]=center->get(2).asDouble();
89  }
90 
91  Bottle *orientation=estimated_superq.find("orientation").asList();
92 
93  if (!estimated_superq.find("orientation").isNull())
94  {
95  Vector axis(4,0.0);
96  axis[0]=orientation->get(0).asDouble(); axis[1]=orientation->get(1).asDouble(); axis[2]=orientation->get(2).asDouble(); axis[3]=orientation->get(3).asDouble();
97  object.setSubvector(8,dcm2euler(axis2dcm(axis)));
98  }
99 
100  readSuperq("hand",hand,11,this->rf);
101 
102  if (left_or_right=="both")
103  {
104  readSuperq("hand1",hand1,11,this->rf);
105  }
106 
107  graspComp->setPar("left_or_right", hand_str);
108  graspComp->run();
109  graspComp->getSolution(hand_str);
110 
111  yInfo()<<" [GraspingModule]: Complete solution "<<complete_sol.toString();
112 
113  t_grasp=graspComp->getTime();
114 
115  graspVis->left_or_right=hand_str;
116 
117  readSuperq("hand",graspVis->hand,11,this->rf);
118 
119  executed_var=false;
120 
121  if (look_object=="on")
122  graspVis->look_object=true;
123 
124  return complete_sol;
125 }

575 /****************************************************************/
576 bool GraspingModule::updateModule()
577 {
578  //LockGuard lg(mutex);
579 
580  if (visualization)
581  {
582  if (times_vis.size()<10)
583  times_vis.push_back(graspVis->getTime());
584  else if (times_vis.size()==10)
585  {
586  for (size_t i=0; i<times_vis.size(); i++)
587  {
588  t_vis+=times_vis[i];
589  }
590  t_vis=t_vis/times_vis.size();
591  times_vis.push_back(0.0);
592  }
593  else
594  times_vis.clear();
595  }
596 
597  if (executed==false)
598  {
599  graspExec->getPoses(complete_sol);
600  executed_var=executed=graspExec->executeTrajectory(hand_to_move);
601 
602  yDebug()<<"executed_var"<<executed_var;
603  }
604 
605  if (save_poses && (graspComp->count_file == graspComp->count_file_old))
606  saveSol(complete_sol);
607 
608  return true;
609 }
